\begin{frame}[fragile]
\frametitle{Vi kan lage arrays og matriser ved å liste alle elementene}

{\alrt \verb+A = [1 3 4; 2 3 1; 2 1 -1]+}
$$A = \begin{pmatrix}
	1 & 3 & 4 \\
	2 & 3 & 1 \\ 
	2 & 1 & -1
\end{pmatrix}$$

{\alrt \verb+x = [1 3 4 2]+}
$$\bt{x} = \begin{pmatrix}
1 & 3 & 4 & 2
\end{pmatrix}$$

{\alrt \verb+y = [1; 3; 4; 2]+}
$$\bt{y} = \begin{pmatrix}
1 \\ 3 \\ 4 \\ 2
\end{pmatrix}$$

\end{frame}

\begin{frame}[fragile]
\frametitle{Det finnes også innebygde funksjoner som lager gitte matriser}

{\alrt \verb+A = zeros(2,4)+}
$$A = \begin{pmatrix}
0 & 0 & 0 & 0 \\
0 & 0 & 0 & 0 
\end{pmatrix}$$

{\alrt \verb+A = ones(1,3)+}
$$A = \begin{pmatrix}
1 & 1 & 1 
\end{pmatrix}$$

{\alrt \verb+A = eye(3,3)+}
$$A = \begin{pmatrix}
1 & 0 & 0 \\
0 & 1 & 0 \\
0 & 0 & 1
\end{pmatrix}$$
\end{frame}

\begin{frame}[fragile]
\frametitle{Det finnes også innebygde funksjoner som lager gitte matriser}

{\alrt \verb+A = rand(4,4)+}
$$A = \begin{pmatrix}
0.8147 & 0.6324 & 0.9575 & 0.9572 \\
0.9058 & 0.0975 & 0.9649 & 0.4854 \\
0.1270 & 0.2785 & 0.1576 & 0.8003 \\ 
0.9134 & 0.5469 & 0.9706 & 0.1419
\end{pmatrix}$$
\end{frame}

\begin{frame}[fragile]
\frametitle{Vi kan behandle bestemte matriseelementer ved å indeksere}
$$A = \begin{pmatrix}
A_{11} & A_{12} & A_{13} \\ A_{21} & A_{22} & A_{23} \\ A_{31} & A_{32} & A_{33}
\end{pmatrix}$$

Vi velger element $A_{ij}$ ved å indeksere {\alrt \verb+A(i,j)+}. Matlab begynner å telle fra 1.
\end{frame}

\begin{frame}[fragile]
\frametitle{Vi kan behandle bestemte matriseelementer ved å indeksere}

{\alrt \verb+A = [2 3 1; 0 -4 3; 1 2 -1]+}
$$A = \begin{pmatrix}
2 & 3 & 1 \\ 0 & -4 & 3 \\ 1 & 2 & -1
\end{pmatrix}$$

\begin{itemize}
	\item Hva er {\alrt \verb+A(2,1) = 0+}?
	\item Hva er {\alrt \verb!A(1,1) + A(2,2) + A(3,3)!}?
	\item Hva gjør kommandoen {\alrt \verb+A(3,2) = 0+}?
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Pass på indeksering, det kan fort bli feil}
{\alrt \verb+A = [2 3 1; 0 -4 1; 3 2 2]+}
$$A = \begin{pmatrix}
	2 & 3 & 1 \\ 0 & -4 & 1 \\ 3 & 2 & 2
\end{pmatrix}$$

\noindent\makebox[\linewidth]{\rule{\paperwidth}{0.4pt}}

{\alrt \verb!A(3,2)+A(1,4)!} \\
{\hspace{0.5cm}$\Rightarrow \quad$  \color{DarkRed} Error using  * 
Inner matrix dimensions must agree.}

\noindent\makebox[\linewidth]{\rule{\paperwidth}{0.4pt}}

{\alrt \verb!A(1,4)=4!} \\
$$A = \begin{pmatrix}
	2 & 3 & 1 & 4\\ 0 & -4 & 1 & 0 \\ 3 & 2 & 2 & 0
\end{pmatrix}$$


\end{frame}




\begin{frame}[fragile]
\frametitle{Matlab tolker operasjoner som vektor- og matrise\-operasjoner}

{\alrt \verb+A = [2, 3, 1; 0, 1, 2; 1, -1, 0]+} \\
{\alrt \verb+x = [1; 0; 3]+} \\
$$A = \begin{pmatrix}
	2 & 3 & 1 \\
	0 & 1 & 2 \\ 
	1 & -1 & 0
\end{pmatrix}, \qquad x = \begin{pmatrix}
1 \\ 0 \\ 3 
\end{pmatrix}.$$
{\alrt \verb+y = A*x+}
$$\bt{y} = \begin{pmatrix}
	2 & 3 & 1 \\
	0 & 1 & 2 \\ 
	1 & -1 & 0 \end{pmatrix}\begin{pmatrix}
1 \\ 0 \\ 3 
\end{pmatrix}
=
\begin{pmatrix}
5 \\ 6 \\ 1
\end{pmatrix}.$$
\end{frame}



\begin{frame}[fragile]
\frametitle{Vi kan også regne komponentvis}

{\alrt \verb+A = [2, 3, 1; 0, 1, 2]+} \\
{\alrt \verb+x = [1, -1, 2; 0, 1, 3]+} \\
$$A = \begin{pmatrix}
	2 & 3 & 1 \\
	0 & 1 & 2 
\end{pmatrix}, \qquad \bt{x} = \begin{pmatrix}
1 & 1 & 2 \\ 0 & 6 & 3  
\end{pmatrix}.$$
{\alrt \verb+y = A.*x+}
$$\bt{y} = \begin{pmatrix}
	2\times1 & 3\times1 & 1\times2 \\
	0\times0 & 1\times6 & 2\times 3 
\end{pmatrix}
=
\begin{pmatrix}
2 & 3 & 2 \\ 0 & 6 & 6
\end{pmatrix}.$$
Merk at {\alrt \verb+*+} operasjonen betyr matrise-ganger-vektor, mens {\alrt \verb+.*+} betyr komponentvis multiplikasjon.
\end{frame}

\begin{frame}[fragile]
\frametitle{Man bør være litt forsiktig med hvilken operator man bruker}
{\alrt \verb+A = [2 3; 1 0]+} \\
$$A = \begin{pmatrix}
	2 & 3  \\
	1 & 0 
\end{pmatrix}.$$
{\alrt \verb+B = A^3+} 
$$
B = \begin{pmatrix}
	2 & 3  \\ 	1 & 0 
\end{pmatrix}\begin{pmatrix}
	2 & 3  \\ 1 & 0 
\end{pmatrix}\begin{pmatrix}
	2 & 3  \\ 1 & 0 
\end{pmatrix} = \begin{pmatrix}
	20 & 21 \\ 7 & 6
\end{pmatrix}.$$
{\alrt \verb+C = A.^3+} 
$$
C = \begin{pmatrix}
	2^3 & 3^3  \\ 	1^3 & 0^3 
\end{pmatrix} = \begin{pmatrix}
	8 & 27  \\ 	1 & 0 
\end{pmatrix}.$$

\end{frame}


\begin{frame}[fragile]
\frametitle{Dimensjoner må stemme når vi gjør matriseoperasjoner}
Matlab tolker egentlig alle arrays som matriser, og gjør dermed forskjellig på kolonne- og radvektorer. 

\vspace{0.2cm}

{\alrt \verb+x = [2 3 1 0]+} \\
{\alrt \verb+y = [1 1 2 4]+} \\
$$\bt{x} = \begin{pmatrix}
	2 & 3 & 1 & 0
\end{pmatrix}, \qquad \bt{y} = \begin{pmatrix}
1 & 1 & 2 & 4
\end{pmatrix}.$$
{\alrt \verb+x*y+} \\ \vspace{0.4cm}
{\hspace{0.5cm}$\Rightarrow \quad$  \color{DarkRed} Error using  * 
Inner matrix dimensions must agree.}

Vi kan ikke ta matrisproduktet av to radvektorer.
\end{frame}

\begin{frame}[fragile]
\frametitle{Dimensjoner må stemme når vi gjør matriseoperasjoner}

For å ta matriseproduktet av to vektorer, må vi ha en radvektor og en kolonnevektor.
\vspace{0.2cm}

{\alrt \verb+x = [2 3 1 0]+} \\
{\alrt \verb+y = [1; 1; 2; 4]+} \\
$$\bt{x} = \begin{pmatrix}
	2 & 3 & 1 & 0
\end{pmatrix}, \qquad \bt{y} = \begin{pmatrix}
1 \\ 1 \\ 2 \\ 4
\end{pmatrix}.$$
{\alrt \verb+x*y+} \\ \vspace{0.4cm}
{\hspace{0.5cm}$\Rightarrow \quad$ 7.}
\end{frame}

\begin{frame}[fragile]
\frametitle{Matriser kan enkelt transponeres}

{\alrt \verb+A = [2 3 1; 0 1 2; 3 0 1]+} \\
{\alrt \verb+B = A'+} \\
$$A = \begin{pmatrix}
	2 & 3 & 1 \\ 0 & 1 & 2 \\ 3 & 0 & 1
\end{pmatrix}, \qquad B = \begin{pmatrix}
	2 & 0 & 3 \\ 3 & 1 & 0 \\ 1 & 2 & 1
\end{pmatrix}.$$

\noindent\makebox[\linewidth]{\rule{\paperwidth}{0.4pt}}

Vi kan også transponere vektorer \\
{\alrt \verb+x = [2 3 1]+} \\
{\alrt \verb+y = x'+} \\
$$\bt{x} = \begin{pmatrix}
	2 & 3 & 1 
\end{pmatrix}\qquad \bt{y} = \begin{pmatrix}
	2 \\ 3 \\ 1 
\end{pmatrix}.$$
\end{frame}

\begin{frame}[fragile]
\frametitle{Vi kan regne ut determinanten av en matrise og radredusere den}

{\alrt \verb+A = [2 3 1; 0 1 2; 3 0 1];+} \\
{\alrt \verb+det(A)+} \\
{\hspace{0.5cm}$\Rightarrow \quad$ 17.}

\noindent\makebox[\linewidth]{\rule{\paperwidth}{0.4pt}}

{\alrt \verb+A = [3 3 1; 6 1 2; 3 2 1]+} \\
{\alrt \verb+B = rref(A)+} \\
$$A = \begin{pmatrix}
	3 & 3 & 1 \\ 6 & 1 & 2 \\ 3 & 2 & 1 
\end{pmatrix}, \qquad B = \begin{pmatrix}
	1.0000 & 0 & 0.3333 \\ 0 & 1.000 & 0 \\ 0 & 0 & 0 
\end{pmatrix}.$$
\end{frame}


\begin{frame}[fragile]
\frametitle{Matlab kan finne den inverse av en matrise}

{\alrt \verb+A = [2 3 1; 0 1 2; 3 0 1]+} \\
{\alrt \verb+B = inv(A)+} 
$$A = \begin{pmatrix}
	2 & 3 & 1 \\ 0 & 1 & 2 \\ 3 & 0 & 1
\end{pmatrix}, \qquad B = \begin{pmatrix}
    0.0588 &  -0.1765 &   0.2941 \\
    0.3529 &  -0.0588 &  -0.2353 \\
   -0.1765 &   0.5294 &   0.1176\end{pmatrix}.$$

\noindent\makebox[\linewidth]{\rule{\paperwidth}{0.4pt}}


Vi kan teste ved å regne ut $AA^{-1}$ \\
{\alrt \verb+C = A*inv(A)+} 
$$C = \begin{pmatrix}
    1.0000   &      0  &  0.0000 \\
         0   & 1.0000   &      0 \\
         0   & 0.0000   & 1.0000
\end{pmatrix}.$$
Vi ser at C er identitetsmatrisen. Merk at utksriften blir litt rar fordi matlab regner numerisk.
\end{frame}

\begin{frame}[fragile]
\frametitle{Vi kan nå løse matriseligninger i Matlab}
Ønsker å løse en ligning av typen
$$A\bt{x} = \bt{b}.$$
Løsningen vil være gitt ved
$$\bt{x} = A^{-1}\bt{b}.$$
{\alrt \verb+A = [2 3; 0 -1]+} \\
{\alrt \verb+b = [2; 1]+} \\
{\alrt \verb+x = inv(A)*b+} 
$$A = \begin{pmatrix}
	2 & 3 \\ 0 & -1
\end{pmatrix}, \qquad \bt{b} = \begin{pmatrix}
	2 \\ 1
\end{pmatrix} \quad \Rightarrow \quad \bt{x} = \begin{pmatrix}
	2.5 \\ -1
\end{pmatrix}.$$
\end{frame}


\begin{frame}[fragile]
\frametitle{Egenverdier og egenvektorer finner vi også lett}
{\alrt \verb+A = [2 3 1; 0 -1 4; 2 2 0]+} \\
{\alrt \verb+[u,v] = eig(A)+} \\
$$u = \begin{pmatrix}
    0.7470  &  0.7071  &  0.5067 \\
    0.3997  & -0.7071  & -0.8189 \\
    0.5313  &  0.0000  &  0.2695
\end{pmatrix}.$$
$$v = \begin{pmatrix}
    4.3166  &       0 &        0 \\
         0  & -1.0000 &        0 \\
         0  &       0 &  -2.3166
\end{pmatrix}.$$
\end{frame}
